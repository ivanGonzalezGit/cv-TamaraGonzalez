create a new repository on the command line

echo "# cv-TamaraGonzalez" >> README.md

git add README.md
git commit -m "first commit"
git branch -M main
git remote add origin https://github.com/ivanGonzalezGit/cv-TamaraGonzalez.git
git push -u origin main
